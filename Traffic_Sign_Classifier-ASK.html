def graynorm(xtensor):
    import cv2
    #Normalize to between 0 and 1
    #Using openCV Gray Scale, strips 3 channels to 1, but doesn't add that channel (i.e. 32x32x3 to 32x32):   
    
    retval = np.zeros([xtensor.shape[0],xtensor.shape[1],xtensor.shape[2]])
                                             
    count = 0
    for img in xtensor:
        img = img.astype(dtype='uint8')
        img = cv2.cvtColor(img, cv2.COLOR_RGB2GRAY)
        img = img.astype(dtype='float32')
        #img[:,:] = (img[:,:]-128)/128
        img[:,:] = (img[:,:]/255)
        retval[count] = img
        count += 1

    return retval

def jitter(xdat, ydat, portion = 0.6, intensity = 5):
    import cv2
    # Takes a portion of the data and generates a new set of geometric transformed synthetic data based on percentage
    # Assume square input e.g. 32x32
    # Intensity of jittering (intensity) used to determine how much 'skewing' of the image is applied
    assert (0 < portion <= 1), "Portion of input data set must be greater than 0 and less than or equal to 1"    
    assert len(xdat) ==  len(ydat), "x_input and y_input have different lengths"
    #print('len',len(xdat))   
    
    # Select random number of features and labels
    randomcount = int(len(xdat)*portion)
    idx = np.random.choice(np.arange(len(xdat)), randomcount, replace=False)
    xnew = xdat[idx]
    ynew = ydat[idx]
    
    count = 0
    
    fig, ax = plt.subplots(figsize = (8,4))
    plt.subplot(121)
    plt.imshow(xnew[0]),plt.title('Example Pre-Distortion Input')
    
    for img in xnew:
        rows,cols,ch = img.shape
        pts1 = np.float32([[0,0],
                           [rows,0],
                           [0,cols],
                           [rows,cols]])
        
        # Set Random Jittering using intensity
        xrand, yrand = np.random.uniform(low=-1, high=1.0, size=[2,4])
        
        pts2 = np.float32([[0+int(xrand[0]*intensity),0+int(yrand[0]*intensity)],
                           [rows+int(xrand[1]*intensity),0+int(yrand[1]*intensity)],
                           [0+int(xrand[2]*intensity),cols+int(yrand[2]*intensity)],
                           [rows+int(xrand[3]*intensity),cols+int(yrand[3]*intensity)]])
        
        M = cv2.getPerspectiveTransform(pts1,pts2)
        dst = cv2.warpPerspective(img,M,(32,32))
  
        xnew[count] = dst
        count += 1        
    
    # Debug: Plot the last image to see how it was transformed
    #print('xrand',xrand)
    #print('yrand',yrand)    
    #plt.subplot(121),plt.imshow(img),plt.title('Last Input')
    plt.subplot(122)
    plt.imshow(xnew[0]),plt.title('Example Post-Distortion Output')
    plt.show()
    fig.savefig('outputfigs/PP-Jitter.png')
    
    # Reshape the new dataset to match the input:  
    
    xnew = np.reshape(xnew, (len(xnew),xdat.shape[1],xdat.shape[2],xdat.shape[3]))
    
    #print("xdatshp",xdat.shape)
    #print("ydatshp",ydat.shape)
    #print("xnewshp",xnew.shape)

# Generate a new set of randomly selected images with rotation, default to use 60% of the original data
def random_rotate(xdat, ydat, portion = 0.6, intensity = 0):
    import cv2
    # Takes a portion of the data and generates a new set of geometric transformed synthetic data based on percentage
    # Assume square input e.g. 32x32
    # Intensity of rotation (intensity) used to determine fineness the rotation angle
    # Discard 0 degree or 360 degree rotations as they will create duplicate images that may not be desirable
    
    assert (0 < portion <= 1), "Portion of input data set must be greater than 0 and less than or equal to 1"
    assert (0 <= intensity <= 1),"Intensity for Random Rotation should be between 0 and 1: only 90 at 0, single degrees at 1"
    assert len(xdat) ==  len(ydat), "x_input and y_input have different lengths"
    #print('len',len(xdat))   
    
    # Select random number of features and labels
    randomcount = int(len(xdat)*portion)
    idx = np.random.choice(np.arange(len(xdat)), randomcount, replace=False)
    xnew = xdat[idx]
    ynew = ydat[idx]
    
    #Remove any Ahead Only, and Keep Right & Left signs as they are not rotationally invariant
    #Strip ID's: 35, 38, 39
    #a = np.array(a)
    #b = np.array(b)[ynew != 23].tolist()
    #a = a[a != 0].tolist()
    xnew = xnew[ynew != 35]
    ynew = ynew[ynew != 35]
    
    xnew = xnew[ynew != 38]
    ynew = ynew[ynew != 38]
    
    xnew = xnew[ynew != 39]
    ynew = ynew[ynew != 39]
    
    count = 0
    
    # Debug: Plot the first image to see how it was transformed  
    fig, ax = plt.subplots(figsize = (8,4))
    plt.subplot(121)
    plt.imshow(xnew[0]),plt.title('Example Pre-Rotation Input')
    
    for img in xnew:
        rows,cols,ch = img.shape
        theta = 0
        
        # define a function that returns 90 degrees if intensity = 0, or single degrees if intensity = 1
        multiple = ((1-90)*intensity + 90)
        
        # Set random rotation using intensity, retry if no change (theta = 0 or 360)
        while (theta == 0) or (theta == 360) :
            anglerand = np.random.uniform(low=-1, high=1.0)

            precision = (multiple/360)
            n = (360*anglerand/multiple)        

            #print("multiple: {}\nanglerand: {}\nprecision: {}\nn: {}\n".format(multiple,anglerand,precision,n))

            theta = round(n)

            theta = theta*360*precision        
                
        M = cv2.getRotationMatrix2D((cols/2,rows/2),-theta,1)
        dst = cv2.warpAffine(img,M,(cols,rows))
    
        xnew[count] = dst
        count += 1        
        # Debug:
        #break
    
    # Debug: Plot the first image to see how it was transformed        
    plt.subplot(122)
    plt.imshow(xnew[0]),plt.title('Example Post-Rotation Output')
    plt.show()
    fig.savefig('outputfigs/PP-Rotation.png')
    # Reshape the new dataset to match the input:  
    
    xnew = np.reshape(xnew, (len(xnew),xdat.shape[1],xdat.shape[2],xdat.shape[3]))
    
    # Append the synthetic data to the original input:
    xdat = np.append(xdat, xnew)
    ydat = np.append(ydat, ynew)
    xdat = np.reshape(xdat, (len(ydat),xnew.shape[1],xnew.shape[2],xnew.shape[3]))
        
    return xdat, ydat

        #    print("{} % Complete".format(round(100*i/xdat.shape[0])))
        xdat[i] = exposure.equalize_adapthist(xdat[i])
    
    return xdat

from tensorflow.contrib.layers import flatten

def LeNet(x):    
    # Arguments used for tf.truncated_normal, randomly defines variables for the weights and biases for each layer
    mu = 0
    sigma = 0.1
    
    # SOLUTION: Layer 1: Convolutional. Input = 32x32x1. Output = 28x28x6.
    conv1_W = tf.Variable(tf.truncated_normal(shape=(5, 5, 1, 6), mean = mu, stddev = sigma))
    conv1_b = tf.Variable(tf.zeros(6))
    conv1   = tf.nn.conv2d(x, conv1_W, strides=[1, 1, 1, 1], padding='VALID') + conv1_b    

    # SOLUTION: Activation.
    conv1 = tf.nn.relu(conv1)
    #tf.add_to_collection('conv1_output', conv1)
    
    # Dropout
    conv1 = tf.layers.dropout(conv1, rate=DropoutRate, noise_shape=None, seed=None, training=False, name='Layer1RELU')
    
    # SOLUTION: Pooling. Input = 28x28x6. Output = 14x14x6.
    conv1 = tf.nn.max_pool(conv1, ksize=[1, 2, 2, 1], strides=[1, 2, 2, 1], padding='VALID')
   
    # SOLUTION: Layer 2: Convolutional. Output = 10x10x16.
    conv2_W = tf.Variable(tf.truncated_normal(shape=(5, 5, 6, 16), mean = mu, stddev = sigma))
    conv2_b = tf.Variable(tf.zeros(16))
    conv2   = tf.nn.conv2d(conv1, conv2_W, strides=[1, 1, 1, 1], padding='VALID') + conv2_b
    
    # SOLUTION: Activation.
    conv2 = tf.nn.relu(conv2)
    
    # Dropout
    conv2 = tf.layers.dropout(conv2, rate=DropoutRate, noise_shape=None, seed=None, training=False, name='Layer2RELU')

    # SOLUTION: Pooling. Input = 10x10x16. Output = 5x5x16.
    conv2 = tf.nn.max_pool(conv2, ksize=[1, 2, 2, 1], strides=[1, 2, 2, 1], padding='VALID')

    # SOLUTION: Flatten. Input = 5x5x16. Output = 400.
    fc0   = flatten(conv2)
    
    # SOLUTION: Layer 3: Fully Connected. Input = 400. Output = 120.
    fc1_W = tf.Variable(tf.truncated_normal(shape=(400, 120), mean = mu, stddev = sigma))
    fc1_b = tf.Variable(tf.zeros(120))
    fc1   = tf.matmul(fc0, fc1_W) + fc1_b
    
    # SOLUTION: Activation.
    fc1    = tf.nn.relu(fc1)
    
    # Dropout
    fc1 = tf.layers.dropout(fc1, rate=DropoutRate, noise_shape=None, seed=None, training=False, name='Layer3RELU')

    # SOLUTION: Layer 4: Fully Connected. Input = 120. Output = 84.
    fc2_W  = tf.Variable(tf.truncated_normal(shape=(120, 84), mean = mu, stddev = sigma))
    fc2_b  = tf.Variable(tf.zeros(84))
    fc2    = tf.matmul(fc1, fc2_W) + fc2_b
    
    # SOLUTION: Activation.
    fc2    = tf.nn.relu(fc2)
    
    # Dropout
    fc2 = tf.layers.dropout(fc2, rate=DropoutRate, noise_shape=None, seed=None, training=False, name='Layer4RELU')

    # SOLUTION: Layer 5: Fully Connected. Input = 84. Output = 10.
    fc3_W  = tf.Variable(tf.truncated_normal(shape=(84, n_classes), mean = mu, stddev = sigma))
    fc3_b  = tf.Variable(tf.zeros(n_classes))
    logits = tf.matmul(fc2, fc3_W) + fc3_b
    return logits

saver = tf.train.Saver()

def evaluate(X_data, y_data):
    num_examples = len(X_data)
    total_accuracy = 0
    sess = tf.get_default_session()
    for offset in range(0, num_examples, BATCH_SIZE):
        batch_x, batch_y = X_data[offset:offset+BATCH_SIZE], y_data[offset:offset+BATCH_SIZE]
        accuracy = sess.run(accuracy_operation, feed_dict={x: batch_x, y: batch_y})
        total_accuracy += (accuracy * len(batch_x))

def scale_images_32(*args):
    import math
    ## Scale the input image to 32x32, crop if not square
    for img in args:
        if (img.shape[0] != img.shape[1]):
                (h, w) = img.shape[:2]
                mindim = min(h,w)
                #print(mindim)
                cx = w / 2
                cy = h / 2
                # Crop at center using minimum dimension of image:
                img = img[math.ceil(cy-mindim/2):math.floor(cy+mindim/2),math.ceil(cx-mindim/2):math.floor(cx+mindim/2)]
        #print("imshp",img.shape)               
        img = cv2.resize(img,(32, 32), interpolation = cv2.INTER_AREA)
    return img

While neural networks can be a great learning device they are often referred to as a black box. We can understand what the weights of a neural network look like better by plotting their feature maps. After successfully training your neural network you can see what it's feature maps look like by plotting the output of the network's weight layers in response to a test stimuli image. From these plotted feature maps, it's possible to see what characteristics of an image the network finds interesting. For a sign, maybe the inner network feature maps react with high activation to the sign's boundary outline or to the contrast in the sign's painted symbol.

Provided for you below is the function code that allows you to get the visualization output of any tensorflow weight layer you want. The inputs to the function should be a stimuli image, one used during training or a new one you provided, and then the tensorflow variable name that represents the layer's state during the training process, for instance if you wanted to see what the LeNet lab's feature maps looked like for it's second convolutional layer you could enter conv2 as the tf_activation variable.

For an example of what feature map outputs look like, check out NVIDIA's results in their paper End-to-End Deep Learning for Self-Driving Cars in the section Visualization of internal CNN State. NVIDIA was able to show that their network's inner weights had high activations to road boundary lines by comparing feature maps from an image with a clear path to one without. Try experimenting with a similar test to show that your trained network's weights are looking for interesting features, whether it's looking at differences in feature maps from images with or without a sign, or even what feature maps look like in a trained network vs a completely untrained one on the same sign image.
